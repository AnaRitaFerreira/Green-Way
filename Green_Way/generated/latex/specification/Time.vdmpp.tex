\begin{vdmpp}[breaklines=true]
class Time
instance variables

public timer: nat1;
public month: nat1;
public year: nat1;

operations

(*@
\label{Diff:10}
@*)
public static Diff : Time * Time ==> real 
Diff (time1,time2) == (
  return time1.timer - time2.timer;
);

(*@
\label{Time:15}
@*)
public Time : nat1 ==> Time
  Time(t) == (
   month := 1;
   year := 2015;
   timer := t;
  return self;
 );
 
functions

(*@
\label{compareTimes:25}
@*)
 public compareTimes : nat1 * nat1 -> int
  compareTimes(time1, time2) == (
  time1- time2
 );
 
end Time
\end{vdmpp}
